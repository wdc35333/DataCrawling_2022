# 확률적 경사하강법을 이용한 문장 유사도 구하기
from sklearn.linear_model import SGDClassifier
from sklearn.feature_extraction.text import TfidfVectorizer

corpus = [

  '먹고 싶은 사과',
  '먹고 싶은 바나나',
  '길고 노란 바나나 바나나',
  '저는 과일이 좋아요'
   
]
tfidfv = TfidfVectorizer().fit(corpus)

X=tfidfv.fit_transform(corpus)
y=[1,2,3,4]

model=SGDClassifier(loss='perceptron')
model.fit(X,y)

newX=tfidfv.transform(['사과가 좋아요','좋은 과일은'])
pred=model.predict(newX)
pred    # 사과가좋아요는 4번문장'저는과일이 좋아요'와 유사
        # 좋은 과일은 은  3번문장 과 유사함을 표시함.
